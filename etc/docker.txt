# For build and run manually
docker build -t edge-service . (in the same folder as Dockerfile)
docker run -p 8080:8080 edge-service

# Pass properties to Spring via Docker as environment variables
docker run -p 8080:8080 -e swagger.title="Hello World" -e swagger.contact-name="Bob" edge-service:1.0-SNAPSHOT
docker run -p 8080:8080 -e SWAGGER_TITLE="Hello World" -e swagger.contact-name="Bob" edge-service:1.0-SNAPSHOT

# Run and listening on debug port 5005
docker run -p 8080:8080 -p 5005:5005 -e JAVA_ARGS="-agentlib:jdwp=transport=dt_socket,server=y,suspend=n,address=*:5005" -e SPRING_PROFILES_ACTIVE=default edge-service:1.0-SNAPSHOT

# Run command inside container (attach on the log file with tail)
docker container exec CONTAINER tail -F logs/edge-service.log
